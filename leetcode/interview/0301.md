```
三合一。描述如何只用一个数组来实现三个栈。

你应该实现push(stackNum, value)、pop(stackNum)、isEmpty(stackNum)、peek(stackNum)方法。stackNum表示栈下标，value表示压入的值。

构造函数会传入一个stackSize参数，代表每个栈的大小。

示例1:
输入：
["TripleInOne", "push", "push", "pop", "pop", "pop", "isEmpty"]
[[1], [0, 1], [0, 2], [0], [0], [0], [0]]
输出：
[null, null, null, 1, -1, -1, true]
说明：当栈为空时`pop, peek`返回-1，当栈满时`push`不压入元素。

示例2:
输入：
["TripleInOne", "push", "push", "push", "pop", "pop", "pop", "peek"]
[[2], [0, 1], [0, 2], [0, 3], [0], [0], [0], [0]]
输出：
[null, null, null, null, 2, 1, -1, -1]
```

```
class TripleInOne {
    /**
     * @param Integer $stackSize
     */
    public $arr = [];
    public $stack_size;
    function __construct($stackSize) {
        $this->stack_size = $stackSize;
        $this->arr[0] = [];
        $this->arr[1] = [];
        $this->arr[2] = [];
    }

    /**
     * @param Integer $stackNum
     * @param Integer $value
     * @return NULL
     */
    function push($stackNum, $value) {
        if (count($this->arr[$stackNum]) < $this->stack_size) {
            array_push($this->arr[$stackNum], $value);
        }
    }

    /**
     * @param Integer $stackNum
     * @return Integer
     */
    function pop($stackNum) {
        if ($this->isEmpty($stackNum)) {
            return -1;
        } else {
            return array_pop($this->arr[$stackNum]);
        }

    }

    /**
     * @param Integer $stackNum
     * @return Integer
     */
    function peek($stackNum) {
        if ($this->isEmpty($stackNum)) {
            return -1;
        } else {
            return end($this->arr[$stackNum]);
        }
    }

    /**
     * @param Integer $stackNum
     * @return Boolean
     */
    function isEmpty($stackNum) {
        return empty($this->arr[$stackNum]);
    }
}
```
