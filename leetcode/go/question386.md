```
给你一个整数 n ，按字典序返回范围 [1, n] 内所有整数。

你必须设计一个时间复杂度为 O(n) 且使用 O(1) 额外空间的算法。

 

示例 1：

输入：n = 13
输出：[1,10,11,12,13,2,3,4,5,6,7,8,9]
示例 2：

输入：n = 2
输出：[1,2]
 

提示：

1 <= n <= 5 * 104
```

```
func lexicalOrder(n int) []int {
    res := make([]int, 0)
    i := 0
    j := 1
    for i < n {
        res = append(res, j)
        if j * 10 <= n {
            j = j * 10
        } else {
            for j % 10 == 9 || j + 1 > n {
                j /= 10
            }
            j++
        }
        i++
    }
    return res
}
```
